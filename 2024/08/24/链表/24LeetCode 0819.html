148.排序链表

class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head:return None

        p=head
        node=[]
        while p:
            node.append([p,p.val])
            p=p.next
        
        node.sort(key=lambda x:x[1])
        print(node)

        p=node[0][0]
        for _ in range(1,len(node)):
            p.next=node[_][0]
            p=p.next

        return node[0][0]
